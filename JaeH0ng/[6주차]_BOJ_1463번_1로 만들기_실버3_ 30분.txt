#include <iostream>
#include <algorithm>
using namespace std;


int main(){
    int n;
    cin >> n;
    int DP[1000001];
    DP[0]=0;
    DP[1]=0;
    DP[2]=1;
    DP[3]=1;
    for(int i=4; i<=n; i++){
        if(i%3!=0 && i%2!=0){
            DP[i] = DP[i-1]+1;
        }else if(i%3==0 && i%2==0){
            DP[i]=min(DP[i/3]+1, DP[i/2]+1);
        }else if(i%3==0){
            DP[i]=min(DP[i/3]+1, DP[i-1]+1);
        }else if(i%2==0){
            DP[i]=min(DP[i/2]+1, DP[i-1]+1);
        }
    }
    cout << DP[n]<< endl;
    
    return 0;
}
